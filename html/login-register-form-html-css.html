                    let container = document.querySelector('.container');
let registerCallBtn = document.querySelector("#register-call-btn");
let loginCallBtn = document.querySelector("#login-call-btn");

registerCallBtn.addEventListener('click', ()=> {
    container.style.transform = 'rotateY(180deg)';
});

loginCallBtn.addEventListener('click', ()=> {
    container.style.transform = 'rotateY(0deg)';
});
